667 double RateSmoothAddSample (RateSmoothData * rsd, double rate, double time)
668 {
669  int tmp;
670 
671  rsd->sample_[rsd->tail_].time_ = time;
672  rsd->sample_[rsd->tail_].rate_ = rate;
673 
674  if (rsd->head_ == rsd->tail_) { /* was empty */
675  rsd->tail_ = (rsd->tail_ + 1) % RATE_SMOOTH_RING_SIZE;
676  rsd->totalData_ = 0;
677  return rate;
678  }
679 
680  rsd->tail_ = (rsd->tail_ + 1) % RATE_SMOOTH_RING_SIZE;
681  tmp = (rsd->tail_ + RATE_SMOOTH_RING_SIZE - 2) % RATE_SMOOTH_RING_SIZE;
682  rsd->totalData_ += rate * (time - rsd->sample_[tmp].time_);
683 
684  while (time - rsd->sample_[rsd->head_].time_ > rsd->span_ + SMALL_FLOAT
685  || rsd->tail_ == rsd->head_) {
686  tmp = (rsd->head_ + 1) % RATE_SMOOTH_RING_SIZE;
687  rsd->totalData_ -=
688  (rsd->sample_[tmp].time_ - rsd->sample_[rsd->head_].time_)
689  * rsd->sample_[tmp].rate_;
690  rsd->head_ = tmp;
691  }
692 
693  if ((rsd->head_ + 1) % RATE_SMOOTH_RING_SIZE == rsd->tail_) {
694  /* has only one recod */
695  rsd->totalData_ = 0;
696  return rate;
697  }
698 
699  return rsd->totalData_ / (time - rsd->sample_[rsd->head_].time_);
700 }
