158 void PgmccSrcAckProc (Agent * agent, Packet * p)
159 {
160  PgmccSrcInfo * psi = (PgmccSrcInfo *) agent->info_;
161  PgmccAck * ack = (PgmccAck *) & (p->type.pa_);
162  double deltaW;
163  uint32 lead, bitmask;
164  int delta;
165  uint32 newAcks;
166  int total, missing;
167  uint32 l;
168 
169  ReschedTimer (agent, PGMCC_SRC_TIMER_CC, PGMCC_SRC_CC_TIMEOUT,
170  & (psi->ccTimerEvnt_));
171 
172  lead = ack->rxwLead_;
173  bitmask = ack->bitmask_;
174 
175  if (p->srcAddr_ == psi->ackerAddr_) {
176  psi->ackerMrtt_ = psi->txwLead_ - lead;
177  psi->ackerLoss_ = psi->ackerMrtt_ * psi->ackerMrtt_ * ack->rxLoss_;
178  }
179 
180  /*
181  * using bitmasks, determine new acks carried by this pkt.
182  * ack_bitmask_ is the old state, bitmask is the one in the ack.
183  * This section is necessary because we dont have cumulative acks
184  */
185  delta = (int) (lead - psi->ackLead_);
186  if (delta > 32) // very new, reset old state
187  psi->ackBitMask_ = 0;
188  else if (delta > 0)
189  psi->ackBitMask_ <<= delta;
190  else if (delta < -32)
191  bitmask = 0;
192  else
193  bitmask <<= -delta;
194 
195  // compute the new acks
196  newAcks = bitmask & ~ (psi->ackBitMask_);
197  // and update state
198  psi->ackBitMask_ |= bitmask ;
199  if (lead > psi->ackLead_) psi->ackLead_ = lead;
200 
201  // compute total acks in this packet
202  for (total = 0, l = newAcks; l; l >>= 1) {
203  if (l & 1) ++ total;
204  }
205 
206  // count missing packets
207  for (missing = 0 , l = ~ (psi->ackBitMask_); l; l >>= 1) {
208  if (l & 1) ++ missing;
209  }
210 
211  if (total == 0) {
212  // fully duplicate acks
213  return ;
214  }
215 
216  // Do not count dups for ACKs within an rtt from prev. cong.
217  if (lead <= psi->ignoreCong_) {
218  missing = 0 ;
219  psi->ackBitMask_ = ~0;
220  }
221 
222  if (missing == 0) {
223  total += psi->dupAcks_ ; // recover previous acks
224  psi->dupAcks_ = 0 ;
225  if (psi->token_ < 0)
226  deltaW = 0 ; // do not increase window
227  else if (psi->window_ < PGMCC_SRC_SS_THRESH)
228  deltaW = 1 ; // exp. increase for small windows
229  else
230  deltaW = 1. / psi->window_ ;
231  psi->token_ += total * (1 + deltaW) ;
232  psi->window_ += total * deltaW ;
233  } else {
234  psi->dupAcks_ += total ;
235  if (psi->dupAcks_ >= PGMCC_SRC_DUP_THRESH) {
236  /*
237  * re-sync window estimate with reality.
238  * Since we know the real window, we also know how
239  * many tokens will arrive (if no loss), and how
240  * many we should ignore.
241  */
242  psi->window_ = (psi->txwLead_ - lead + 1 );
243  // mimic TCP slowstart
244  psi->ignoreCong_ = psi->txwLead_;
245  if (psi->token_ < 0)
246  ++ psi->token_;
247  else if (psi->window_ < 4 ) // small window
248  psi->token_ = 1 ;
249  else {
250  psi->window_ /= 2 ;
251  psi->token_ = - psi->window_ + 1;
252  }
253  psi->ackBitMask_ = ~0 ; // not count anymore these losses
254  psi->dupAcks_ = 0 ;
255  } else if (psi->txwLead_ == lead )
256  /*
257  * XXX if there are no flying-packets we forcely send a new
258  * one to avoid a stall
259  */
260  psi->token_ = 1;
261  }
262 
263  PgmccSrcSendPkt (agent);
264 }

390 void PgmccRcvPktHandler (Agent * agent, Packet * p)
391 {
392  PgmccRcvInfo * pri = (PgmccRcvInfo *) agent->info_;
393  PgmccData * data = (PgmccData *) &(p->type.pd_);
394  tw_lp * lp = tw_getlp (agent->nodeId_);
395  uint32 i, sqn;
396  int delta;
397 
398  if (! pri->active_) return;
399 
400  if (p->type_ != PGMCC_DATA) {
401  printf ("PGMCC receiver (%d.%lu) received non-data packet.\n",
402  lp->id, agent->id_);
403  tw_exit (-1);
404  }
405 
406  if (pri->rxwLead_ == (uint32) -1) pri->rxwLead_ = data->sqn_;
407 
408  if (data->ackerAddr_ != (uint32) -1) {
409  pri->rxDoAck_ = (data->ackerAddr_ == _nodeTable[lp->id].addr_);
410  }
411  else {
412  pri->rxDoAck_ = 0;
413  PgmccRcvSendPkt (agent, p, PGMCC_NAK);
414  }
415 
416  // update the bitmask
417  sqn = data->sqn_;
418  delta = (int) (sqn - pri->rxwLead_);
419  if (sqn > pri->rxwLead_) {
420  if (delta > 31)
421  pri->ackBitMask_ = 0 ;
422  else
423  pri->ackBitMask_ <<= delta;
424  pri->ackBitMask_ |= 1 ;
425  }
426  else { /* old, out of sequence ? */
427  delta = -delta ;
428  if (delta < 31) pri->ackBitMask_ |= (1 << delta) ;
429  }
430 
431  // update rx_loss_, assuming no out-of-order packets
432  if (sqn > pri->rxwLead_) {
433  pri->rxLoss_ *= pow (PGMCC_RCV_WEIGHT, sqn - pri->rxwLead_);
434 
435  for (i = sqn - 1 ; i > pri->rxwLead_; -- i) {
436  pri->rxLoss_ += pow (PGMCC_RCV_WEIGHT, i - pri->rxwLead_)
437  * (1. - PGMCC_RCV_WEIGHT);
438  }
439  }
440 
441  if (sqn > pri->rxwLead_ + 1) PgmccRcvSendPkt (agent, p, PGMCC_NAK);
442  if (sqn > pri->rxwLead_) pri->rxwLead_ = sqn;
443  if (pri->rxDoAck_) PgmccRcvSendPkt (agent, p, PGMCC_ACK);
444 }
