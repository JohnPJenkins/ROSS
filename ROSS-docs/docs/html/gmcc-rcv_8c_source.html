617 int GmccRcvJump (Agent * agent, double timestamp, int reason)
618 {
619  GmccRcvInfo * gri = (GmccRcvInfo *) agent->info_;
620  tw_lp * lp = tw_getlp (agent->nodeId_);
621  double now = tw_now (lp);
622  int i, availLayer, moreCg;
623 
624  if (gri->joinResult_ == 0) {
625  return 0; /* No result of last join yet */
626  }
627 
628  /* Check whether all joined layers have CR and have got
629  * enough samples. If not, don't join new layer */
630  availLayer = gri->topJoinedLayer_ + 1;
631  moreCg = 0;
632  for (i = 0; i <= gri->topJoinedLayer_; ++ i) {
633  if (! gri->activeLayer_[i]) {
634  availLayer = i;
635  continue;
636  }
637  if (gri->cr_[i] == (uint32) -1 || gri->crTafAvg_[i] <= 0
638  || gri->crTafDev_[i] < 0) {
639  return 0; /* This layer does not have CR yet */
640  }
641  if (gri->moreCgNum_[i] > 0 || gri->cr_[i] == _nodeTable[lp->id].addr_) {
642  if (++ moreCg == 1) { /* Limit the number of layers in which
643  * the receiver can be the most congested
644  * for join */
645  return 0;
646  }
647  }
648  }
649 
650  if (availLayer >= gri->layerNum_) {
651  return 0; /* No free layer found */
652  }
653 
654  /* If the layer to join is already active but does not have a CR yet,
655  * don't join */
656  if (availLayer > gri->topJoinedLayer_) {
657  if (gri->activeLayer_[availLayer] && gri->cr_[i] == (uint32) -1) {
658  return 0;
659  }
660  gri->topJoinedLayer_ = gri->lastJoinedLayer_ = availLayer;
661  }
662  else {
663  gri->lastJoinedLayer_ = availLayer;
664  }
665 
666  /* Layer i is the lowest inactive layer this receiver is in
667  * or the lowest active layer this receiver has not joined
668  * If not in this layer, join it */
669  if (! gri->joinedLayer_[gri->lastJoinedLayer_]) {
670  JoinGroup (lp->id, gri->grpAddrs_[gri->lastJoinedLayer_]);
671  RegisterAgentCopy (agent, gri->grpAddrs_[gri->lastJoinedLayer_]);
672  gri->joinedLayer_[gri->lastJoinedLayer_] = 1;
673  }
674  gri->joinResult_ = 0;
675 
676  /* Reinitialize statistic for this layer */
677  GmccRcvInitLayer2 (agent, gri->lastJoinedLayer_, gri->lastJoinedLayer_ - 1);
678 
679  /* "Less congested" statistics is not valid any more for any layer */
680  for (i = 0; i < gri->layerNum_; ++ i) {
681  gri->lessCgNum_[i] = gri->lessCgNum2_[i] = 0;
682  gri->moreCgNum_[i] = 0; /* To prevent prompt leave after join */
683  gri->lastResetTime_[i] = now;
684  gri->sampleNum_[i] = gri->ovsdSampleNum_[i] = 0;
685  }
686 
687  /* Send a CI after joining a new layer to let the source know */
688  GmccRcvSendCiPkt (agent, timestamp, gri->lastJoinedLayer_, 0, 0, 0);
689  return 1;
690 }

694 {
695  GmccRcvInfo * gri = (GmccRcvInfo *) agent->info_;
696  tw_lp * lp = tw_getlp (agent->nodeId_);
697  double now = tw_now (lp);
698  int i, j;
699 
700  /* Finr top joined active layer */
701  for (i = gri->topJoinedLayer_; i > 0; -- i) {
702  if (gri->activeLayer_[i]) break;
703  }
704 
705  if (i == 0) {
706  return 0;
707  }
708 
709  j = i - 1;
710  /* Leave the top joined active layer and all joined layer above */
711  for (; i <= gri->topJoinedLayer_; ++ i) {
712  LeaveGroup (lp->id, gri->grpAddrs_[i]);
713  UnregisterAgentCopy (lp->id, gri->grpAddrs_[i]);
714  gri->joinedLayer_[i] = 0;
715  GmccRcvInitLayer (agent, i);
716  }
717  gri->topJoinedLayer_ = j;
718

720  for (i = 0; i < gri->layerNum_; ++ i) {
721  gri->moreCgNum_[i] = 0;
722  gri->lessCgNum_[i] = gri->lessCgNum2_[i] = 0; /* To prevent prompt join
723  * after leave */
724  gri->lastResetTime_[i] = now;
725  gri->sampleNum_[i] = gri->ovsdSampleNum_[i] = 0;
726  }
727 
728  return 1;

764 void GmccRcvUpdateCgCmp (Agent * agent, int i)
765 {
766  GmccRcvInfo * gri = (GmccRcvInfo *) agent->info_;
767  double mean, dev, d1, d2, d3;
768  double alpha, beta;
769  int n;
770 
771  /*
772  * Update less/more-congested-than-CR status for this layer
773  * Same as in RecvDataPkt ()
774  */
775  if (gri->cr_[i] == _nodeTable[agent->nodeId_].addr_) {
776  gri->moreCgNum_[i] = gri->lessCgNum_[i] = gri->lessCgNum2_[i] = 0;
777  return;
778  }
779 
780  if (! gri->activeLayer_[i] || gri->cr_[i] == (uint32) -1
781  || gri->crTafAvg_[i] < 0 || gri->tafEst_[i].sampleNum_ == 0) {
782  return;
783  }
784 
785  alpha = GMCC_RCV_CHANGE_CR_FACTOR2;
786  beta = GMCC_RCV_JUMP_FACTOR2;
787  n = gri->tafEst_[i].sampleNum_;
788  mean = gri->tafEst_[i].mean_;
789  dev = gri->tafEst_[i].dev_;
790  d1 = sqrt (dev * dev / n
791  + gri->crTafDev_[i] * gri->crTafDev_[i] / GMCC_RCV_SAMPLE_SIZE);
792  d1 = gri->crTafAvg_[i] + GMCC_RCV_MORE_CG_FACTOR * d1;
793  d2 = sqrt (beta * beta * dev * dev / n
794  + gri->crTafDev_[i] * gri->crTafDev_[i] / GMCC_RCV_SAMPLE_SIZE);
795  d2 = beta * mean + GMCC_RCV_JUMP_FACTOR * d2;
796  d3 = sqrt (dev * dev / n
797  + alpha * alpha * gri->crTafDev_[i] * gri->crTafDev_[i]
798  / GMCC_RCV_SAMPLE_SIZE);
799  d3 = alpha * gri->crTafAvg_[i] + GMCC_RCV_CHANGE_CR_FACTOR * d3;
800 
801  if (gri->crTafAvg_[i] > d2) {
802  ++ gri->lessCgNum_[i];
803  gri->moreCgNum_[i] = 0;
804  return;
805  }
806  else {
807  gri->lessCgNum_[i] = 0;
808  }
809 
810  if (gri->sampleNum_[i] + gri->ovsdSampleNum_[i]
811  < GMCC_RCV_REQD_SAMPLE_SIZE) {
812  return;
813  }
814 
815  if (mean > d1) {
816  ++ gri->moreCgNum_[i];
817  gri->lessCgNum_[i] = 0;
818  return;
819  }
820 
821  gri->moreCgNum_[i] = gri->lessCgNum_[i] = 0;
822 }
