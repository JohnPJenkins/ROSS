32  static long global_id = 0;
33  /* current_id is just the starting ID for this particular iteration */
34  long current_id = global_id;
35 
36  /* Initialize the state of this LP (or master) */
37  num_radios = tw_rand_integer(lp->rng, 1, SRW_MAX_GROUP_SIZE);
38  s->num_radios = num_radios;
39  s->movements = 0;
40  s->comm_fail = 0;
41  s->comm_try = 0;
42 
43  /* Initialize nodes */
44  for (i = 0; i < num_radios; i++) {
45  (s->nodes[i]).node_id = global_id++;
46  (s->nodes[i]).lng = 0.0;
47  (s->nodes[i]).lat = 0.0;
48  (s->nodes[i]).movements = 0;
49  (s->nodes[i]).comm_fail = 0;
50  (s->nodes[i]).comm_try = 0;
51  }
52 
53  /* Priming events */
54  for (i = 0; i < num_radios; i++) {
55  tw_stime ts;
56  srw_msg_data *msg;
57 
58  // GPS event first, as it's the simplest
59  ts = tw_rand_unif(lp->rng);
60  ts = ts * SRW_GPS_RATE;
61 
62  e = tw_event_new(lp->gid, ts, lp);
63  msg = tw_event_data(e);
64  msg->node_id = current_id + i;
65  msg->type = GPS;
66  tw_event_send(e);
67 
68  // Now schedule some movements
69  ts = tw_rand_exponential(lp->rng, SRW_MOVE_MEAN);
70  e = tw_event_new(lp->gid, ts, lp);
71  msg = tw_event_data(e);
72  msg->node_id = current_id + i;
73  msg->type = MOVEMENT;
74  // We have to figure out a good way to set the initial lat/long
75  tw_event_send(e);
76 
77  // Now some communications
78  ts = tw_rand_exponential(lp->rng, SRW_COMM_MEAN);
79  e = tw_event_new(lp->gid, ts, lp);
80  msg = tw_event_data(e);
81  msg->node_id = current_id + i;
82  msg->type = COMMUNICATION;
83  // Something should probably go here as well...
84  tw_event_send(e);
85  }
