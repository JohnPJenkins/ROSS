37 static void sift_down( tw_pq *h, int i )
38 {
39  int n = h->nelems, k = i, j, c1, c2;
40  ELEMENT_TYPE temp;
41 
42  if( n <= 1 ) return;
43 
44  /* Stops when neither child is "strictly less than" parent */
45  do{
46  j = k;
47  c1 = c2 = 2*k+1;
48  c2++;
49  if( c1 < n && KEY(h->elems[c1]) < KEY(h->elems[k]) ) k = c1;
50  if( c2 < n && KEY(h->elems[c2]) < KEY(h->elems[k]) ) k = c2;
51  SWAP( h, j, k, temp );
52  }while( j != k );
53 }

56 static void percolate_up( tw_pq *h, int i )
57 {
58  int n = h->nelems, k = i, j, p;
59  ELEMENT_TYPE temp;
60 
61  if( n <= 1 ) return;
62 
63  /* Stops when parent is "less than or equal to" child */
64  do
65  {
66  j = k;
67  if( (p = (k+1)/2) )
68  {
69  --p;
70  if( KEY(h->elems[k]) < KEY(h->elems[p]) ) k = p;
71  }
72  SWAP( h, j, k, temp );
73  }while( j != k );
74 }

78 {
