489 int ForwardPktOnLink (tw_lp * lp, Link * lnk, Packet * p)
490 {
491  double t, now = tw_now (lp);
492 
493  if (lnk->lastPktOutTime_ <= now) {
494  /* No packet is being sent out. Therefore forward it right now */
495  t = p->size_ * 8 / ((double) lnk->bw_); /* Transmission delay */
496  lnk->lastPktOutTime_ = now + t;
497  p->inPort_ = lnk->nbPort_; /* This will copied to the outgoing packet */
498  return ForwardPacket (lp, tw_getlp (lnk->neighbour_), p, t + lnk->latency_);
499  }
500 
501  if (p->size_ + (lnk->lastPktOutTime_ - now) / 8 * lnk->bw_
502  > lnk->bufSize_ + SMALL_FLOAT) {
503  /* Buffer is full. Discard the packet
504  * The buffered packets include the one being sent except the portion
505  * that has been sent. */
506 
507 #ifdef DEBUG_PKT_DISCARD
508  printf ("%lf : pkt(%d:%lu) at %d discarded\n",
509  tw_now (lp), _addrToIdx[p->srcAddr_], p->od_.sqn_, lp->id);
510  fflush (stdout);
511 #endif
512 
513  return 0;
514  }
515 
516  /* The buffer is not zero, some packet is being sent.
517  * Schedule to forward this packet at a future time */
518  p->inPort_ = lnk->nbPort_; /* This will copied to the outgoing packet */
519  lnk->lastPktOutTime_ += p->size_ * 8 / ((double) lnk->bw_);
520  return ForwardPacket (lp, tw_getlp (lnk->neighbour_), p,
521  lnk->lastPktOutTime_ + lnk->latency_ - now);
522 }
