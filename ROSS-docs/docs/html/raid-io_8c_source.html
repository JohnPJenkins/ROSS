41 void raid_io_eventhandler( IOState* s, tw_bf* cv, MsgData* m, tw_lp* lp )
42 {
43  Event next_event_type;
44  tw_stime this_event_length;
45 
46  // Check the message type
47  switch( m->event_type )
48  {
49  case IO_BUSY:
50  s->mode = BUSY;
51  // Calculate the time to spend in BUSY mode
52  this_event_length = tw_rand_normal_sd( lp->rng, BUSY_TIME,
53  STD_DEV * BUSY_TIME, &(m->rc.rng_calls) );
54  s->ttl_busy += this_event_length;
55  next_event_type = IO_IDLE;
56  break;
57  case IO_IDLE:
58  s->mode = IDLE;
59  // Calculate the time to spend in IDLE mode
60  this_event_length = tw_rand_normal_sd( lp->rng, IDLE_TIME,
61  STD_DEV * IDLE_TIME, &(m->rc.rng_calls) );
62  s->ttl_idle += this_event_length;
63  next_event_type = IO_BUSY;
64  break;
65  default:
66  message_error( m->event_type );
67  }
68 
69  // Generate and send the BUSY message to self
70  raid_io_gen_send( lp->gid, next_event_type, this_event_length, lp );
71  // Generate and send the BUSY message to server
72  raid_io_gen_send( s->m_server, next_event_type, this_event_length, lp );
73 }
