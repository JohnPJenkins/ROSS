39 {

47  j = k;
48  c1 = c2 = 2*k+1;
49  c2++;
50  if( c1 < n && KEY(h->elems[c1]) < KEY(h->elems[k]) ) k = c1;
51  if( c2 < n && KEY(h->elems[c2]) < KEY(h->elems[k]) ) k = c2;
