167  if( lp->id < (number_of_applications / 2) )
168  dest = tw_rand_integer( lp->rng,
169  (number_of_applications / 2),
170  number_of_applications - 1);

172  dest = tw_rand_integer( lp->rng,
173  0,

175 
176  e = tw_event_new(SV->u.application.router->gid, ts, lp);

181  tw_event_send( e );
182 
183 #ifdef TRACE

186  packet->destination->id, SV->u.application.packets_sent_in_burst );
187 #endif /* TRACE */

189  e = tw_event_new(lp->gid, ts, lp);
