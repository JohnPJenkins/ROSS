83 void raid_disk_eventhandler( DiskState* s, tw_bf* cv, MsgData* m, tw_lp* lp )
84 {
85  // Declare time variable
86  tw_stime time;
87  // Check message type
88  switch( m->event_type )
89  {
90  case DISK_FAILURE:
91  time = tw_rand_normal_sd( lp->rng, REPLACE_TIME,
92  STD_DEV * REPLACE_TIME, &(m->rc.rng_calls) );
93  // Generate and send messages
94  raid_disk_gen_send( s->m_controller, DISK_REPLACED, time, lp );
95  raid_disk_gen_send( lp->gid, DISK_REPLACED, time, lp );
96  break;
97  case DISK_REPLACED:
98  time = raid_disk_fail_time( lp );
99  if( ( tw_now( lp ) + time ) < (tw_stime)MAX_HOURS )
100  {
101  // Generate and send messages
102  raid_disk_gen_send( s->m_controller, DISK_FAILURE, time, lp );
103  raid_disk_gen_send( lp->gid, DISK_FAILURE, time, lp );
104 
105  // Modify state
106  cv->c0 = 1;
107  s->num_failures++;
108  }
109  break;
110  default:
111  message_error( m->event_type );
112  }
