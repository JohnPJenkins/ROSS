404 {

406  int i;
407  double ave_interval1, ave_interval2;
408  int ds ;
409  double ave, avg;
410  int factor = 2;
411  double ratio;
412  double min_ratio;
413 
414  // sample[i] counts the number of packets since the i-th loss event
415  // sample[0] contains the most recent sample.
416 
417  //packets (and possibly loss events) are aggregated to form fairsize_'ed
418  // packets
419 
420  for (i = tri->last_sample; i <= tri->maxseq ; i ++) {
421  if (tri->aggregate_packets_) {
422  if (! tri->aggregate_loss_
423  && tri->lossvec_[i % tri->hsz] == TFMCC_RCV_LOST) {
424  // adjust sample if each loss event counts as full packet
425  tri->sum_psize_ += tri->fairsize_;

427  tri->sum_psize_ += tri->sizevec_[i % tri->hsz];
428  }
429 
430  if (tri->sum_psize_ >= 0) {
431  // after received packet ignore the next packets until fairsize_ bytes
432  // were received
433  tri->sum_psize_ -= tri->fairsize_;
434  ++ tri->sample[0];
435  }
436  } else {
437  ++ tri->sample[0];
438  }
439 
440  if (tri->lossvec_[i % tri->hsz] == TFMCC_RCV_NOLOSS
441  || tri->lossvec_[i % tri->hsz] == TFMCC_RCV_LOST) {
442  tri->sum_lsize_ += tri->sizevec_[i % tri->hsz];
443  }
444 
445  if ((tri->aggregate_loss_ && tri->sum_lsize_ >= 0) ||
446  (! tri->aggregate_loss_
447  && tri->lossvec_[i % tri->hsz] == TFMCC_RCV_LOST)) {
448  tri->sum_lsize_ -= tri->fairsize_;

462  else
463  ds = tri->sample_count;
464

466  // no losses yet
467  return 0;

472  ave = TfmccRcvWeightedAvg
473  (agent, 1, ds, 1.0, tri->mult, tri->weights, tri->sample);
474  factor = 2;
475  ratio = (factor * ave) / tri->sample[0];
476  min_ratio = 0.5;
477  if ( ratio < 1.0) {
478  // the most recent loss interval is very large
479  tri->mult_factor_ = ratio;
480  if (tri->mult_factor_ < min_ratio)
481  tri->mult_factor_ = min_ratio;
482  }
483  }

489  // calculations only if this increases the estimated loss
490  // interval.
491  ave_interval2 = TfmccRcvWeightedAvg
492  (agent, 1, ds, tri->mult_factor_, tri->mult, tri->weights, tri->sample);
493  if (ave_interval2 > ave_interval1)
494  ave_interval1 = ave_interval2;
495  if (ave_interval1 < 0.99) {
496  avg = 0;
497  for (i = 0 ; i <= tri->numsamples; i++) {
498  avg += tri->sample[i];
499  }
500  tw_exit (1);
501  }
502 
503  return 1 / ave_interval1;
504 }

508  double *m, double *w, int *sample)
509 {
