652 /*Each MPI LP in this model generates a MPI message until a certain message count is reached.
653 This method
654  (i) keeps generating MPI messages,
655  (ii) breaks a MPI message down to torus packets
656  (iii) sends those packets to the underlying torus node LP */
657 void mpi_msg_send(mpi_process * p,
658  tw_bf * bf,
659  nodes_message * msg,
660  tw_lp * lp)
661 {
662  tw_stime ts;
663  tw_event *e;
664  nodes_message *m;
665  tw_lpid final_dst;
666  int i;
667  bf->c3 = 0;
668  bf->c4 = 0;
669 
670  if(p->message_counter >= injection_limit)
671  {
672  bf->c4 = 1;
673 
674  return;
675  }
676 
677  switch(TRAFFIC)
678  {
679  /* UR traffic picks a random destination from the network */
680  case UNIFORM_RANDOM:
681  {
682  bf->c3 = 1;
683 
684  final_dst = tw_rand_integer( lp->rng, N_nodes, 2 * N_nodes - 1);
685 
686  /* if the random final destination generated is the same as current LP ID then it is possible
687  that the next randomly generated destination is also the same.
688  Therefore if randomly generated destination is the same as source, we use the following
689  calculation to make sure that the source and destinations are different */
690  if( final_dst == lp->gid )
691  {
692  final_dst = N_nodes + (lp->gid + N_nodes/2) % N_nodes;
693  }
694  }
695  break;
696 
697  /* The nearest neighbor traffic works in a round-robin fashion. The first message is sent to the first nearest neighbor
698  of the node, second message is sent to second nearest neighbor and so on. Thats why we use the message counter number
699  to calculate the destination neighbor number (Its between 1 to neighbor-1). In the packet_generate function, we calculate
700  the torus coordinates of the destination neighbor. */
701  case NEAREST_NEIGHBOR:
702  {
703  final_dst = p->message_counter% (2*N_dims_sim);
704  }
705  break;
706 
707  /* The diagonal traffic pattern sends message to the mirror torus coordinates of the current torus node LP. The
708  torus coordinates are not available at the MPI process LP level thats why we calculate destination for this pattern
709  in the packet_generate function. */
710  case DIAGONAL:
711  {
712  final_dst = -1;
713  }
714  }
715  tw_stime base_time = MEAN_PROCESS;
716 
717  for( i=0; i < num_packets; i++ )
718  {
719  // Send the packet out
720  ts = 0.1 + tw_rand_exponential(lp->rng, MEAN_INTERVAL/200);
721  msg->saved_available_time = p->available_time;
722  p->available_time = max( p->available_time, tw_now(lp) );
723  p->available_time += ts;
724 
725  e = tw_event_new( getProcID(lp->gid), p->available_time - tw_now(lp), lp );
726 
727  m = tw_event_data( e );
728  m->type = GENERATE;
729  m->packet_ID = packet_offset * ( lp->gid * num_mpi_msgs * num_packets ) + p->message_counter;
730 
731  p->message_counter++;
732  m->travel_start_time = tw_now( lp ) + ts;
733 
734  if(TRAFFIC == NEAREST_NEIGHBOR || TRAFFIC == DIAGONAL)
735  m->dest_lp = final_dst;
736  else
737  {
738  m->dest_lp = getProcID( final_dst );
739  }
740 
741  m->next_stop = -1;
742  tw_event_send( e );
743  }
744  ts = 0.1 + tw_rand_exponential( lp->rng, MEAN_INTERVAL);
745  e = tw_event_new( lp->gid, ts, lp );
746  m = tw_event_data( e );
747  m->type = MPI_SEND;
748  tw_event_send( e );
749 }
