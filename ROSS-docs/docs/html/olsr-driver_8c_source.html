219 DoCalcRxPower (double txPowerDbm,
220  node_state *s,
221  olsr_msg_data *m)
222 {
223  /*
224  * Friis free space equation:
225  * where Pt, Gr, Gr and P are in Watt units
226  * L is in meter units.
227  *
228  * P Gt * Gr * (lambda^2)
229  * --- = ---------------------
230  * Pt (4 * pi * d)^2 * L
231  *
232  * Gt: tx gain (unit-less)
233  * Gr: rx gain (unit-less)
234  * Pt: tx power (W)
235  * d: distance (m)
236  * L: system loss
237  * lambda: wavelength (m)
238  *
239  * Here, we ignore tx and rx gain and the input and output values
240  * are in dB or dBm:
241  *
242  * lambda^2
243  * rx = tx + 10 log10 (-------------------)
244  * (4 * pi * d)^2 * L
245  *
246  * rx: rx power (dB)
247  * tx: tx power (dB)
248  * d: distance (m)
249  * L: system loss (unit-less)
250  * lambda: wavelength (m)
251  */
252 
253  double sender_lng = m->lng;
254  double sender_lat = m->lat;
255  double receiver_lng = s->lng;
256  double receiver_lat = s->lat;
257 
258  // We have to make sure that everyone is in the same region even though
259  // they may have overlapping x/y coordinates, i.e. a region describes the
260  // local plane of existence for the nodes
261  assert(region(s->local_address) == region(m->originator));
262 
263  double distance = (sender_lng - receiver_lng) * (sender_lng - receiver_lng);
264  distance += (sender_lat - receiver_lat) * (sender_lat - receiver_lat);
265 
266  distance = sqrt(distance);
267 
268  //double distance = a->GetDistanceFrom (b);
269  double m_minDistance = 1.0; // A reasonable default
270  if (distance <= m_minDistance)
271  {
272  return txPowerDbm;
273  }
274  //double m_lambda = 3.0e8 / 2437000000.0; // (2.437 GHz, chan 6)
275  double m_lambda = 0.058; // Stolen from Ken's slides, ~ 5GHz
276  double numerator = m_lambda * m_lambda;
277  double denominator = 16 * M_PI * M_PI * distance * distance;// * m_systemLoss;
278  double pr = 10 * log10 (numerator / denominator);
279  //printf("distance=%fm, attenuation coefficient=%fdB\n", distance, pr);
280  return txPowerDbm + pr;
281 }
