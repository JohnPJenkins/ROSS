37 void raid_server_eventhandler( ServerState* s, tw_bf* cv, MsgData* m, tw_lp* lp )
38 {
39  // Check message type
40  switch( m->event_type )
41  {
42  case IO_IDLE:
43  s->mode = IDLE;
44  break;
45  case IO_BUSY:
46  s->mode = BUSY;
47  break;
48  case REBUILD_START:
49  s->num_controllers_rebuilding++;
50  s->num_controllers_failure--;
51  break;
52  case REBUILD_FINISH:
53  s->num_controllers_good_health++;
54  s->num_controllers_rebuilding--;
55  break;
56  case RAID_FAILURE:
57  s->num_controllers_failure++;
58  s->num_controllers_rebuilding--;
59  break;
60  case DISK_FAILURE:
61  s->num_controllers_failure++;
62  s->num_controllers_good_health--;
63  break;
64  default:
65  message_error( m->event_type );
66  }
67 
68  // Recalculate bandwidth
69  raid_server_calculate_bandwidth( s );
70  // Add to the blocks read/written count
71  if( m->event_type != IO_BUSY )
72  s->num_blocks_wr += m->rc.server_blocks = raid_server_num_blocks( s, lp );
73  // Save current timestamp
74  m->rc.server_timestamp = s->mode_change_timestamp;
75  // Set new timestamp
76  s->mode_change_timestamp = tw_now( lp );
