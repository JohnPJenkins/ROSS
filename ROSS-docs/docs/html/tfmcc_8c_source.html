114 void TfmccSrcPktHandler (Agent * agent, Packet * p)
115 {
116  TfmccSrcInfo * tsi = (TfmccSrcInfo *) agent->info_;
117  tw_lp * lp = tw_getlp (agent->nodeId_);
118  double now = tw_now (lp);
119  TfmccNak * nck = (TfmccNak *) & (p->type.tn_);
120  int cur_prio;
121  int new_clr = 0;
122  double adjusted_rate, tmp_rtt;
123 
124  if (! tsi->active_) return;
125 
126  if (p->type_ != TFMCC_NAK) {
127  printf ("TFMCC source (%d.%lu) received non-nak packet.\n",
128  lp->id, agent->id_);
129  tw_exit (-1);
130  }
131 
132  if (tsi->rate_change_ == TFMCC_SRC_STATE_INIT_RATE) {
133  tsi->rate_change_ = TFMCC_SRC_STATE_SLOW_START;
134  }
135 
136  // if we don't get ANY feedback for some time, cut rate in half
137  ReschedTimer (agent, TFMCC_SRC_TIMER_NO_FDBK,
138  TFMCC_SRC_DEC_NO_REPORT * tsi->max_rtt_, & (tsi->noFdbkTimerEvnt_));
139 
140  // compute receiver's current RTT
141  tmp_rtt = now - nck->timestamp_echo - nck->timestamp_offset;
142  if (tsi->max_rtt_ < tmp_rtt) tsi->max_rtt_ = tmp_rtt;
143 
144  adjusted_rate = nck->rate;
145  if (nck->have_rtt == 0 && tsi->rate_change_ == TFMCC_SRC_STATE_CONG_AVOID) {
146  // recalculate rate with correct RTT
147  adjusted_rate = adjusted_rate / tmp_rtt * tsi->max_rtt_;
148  }
149 
150  if (adjusted_rate < TFMCC_SRC_MIN_RATE) adjusted_rate = TFMCC_SRC_MIN_RATE;
151 
152  // adjust suppression rate by G_FACTOR if non-CLR (and correct feedback round)
153  // before clr_id is updated so that new CLR also contributes to suppression
154  if ((tsi->clr_id != p->srcAddr_)
155  && (tsi->supp_rate_ > (1 - TFMCC_SRC_G_FACTOR) * nck->rate)
156  && (nck->round_id == tsi->round_id)) {
157  tsi->supp_rate_ = (1 - TFMCC_SRC_G_FACTOR) * nck->rate;
158  }
159 
160  // if we are in slow start and we just saw a loss then come out of slow start
161  if (tsi->rate_change_ == TFMCC_SRC_STATE_SLOW_START && nck->have_loss) {
162  tsi->rate_change_ = TFMCC_SRC_STATE_CONG_AVOID;
163  tsi->oldrate_ = tsi->rate_ = adjusted_rate;
164  tsi->clr_id = p->srcAddr_;
165  new_clr = 1;
166  }
167 
168  if (tsi->rate_change_ == TFMCC_SRC_STATE_SLOW_START) {
169  // adjust maxrate
170  if (tsi->clr_id == p->srcAddr_) {
171  tsi->maxrate_ = adjusted_rate;
172  } else if ((nck->receiver_leave == 0)
173  && (adjusted_rate < tsi->maxrate_ || tsi->clr_id == -1)) {
174  // change CLR
175  tsi->maxrate_ = adjusted_rate;
176  tsi->clr_id = p->srcAddr_;
177  new_clr = 1;
178  }
179  TfmccSrcSlowStart (agent);
180  } else if (tsi->rate_change_ == TFMCC_SRC_STATE_CONG_AVOID) {
181  if ((nck->receiver_leave == 0)
182  && (adjusted_rate < tsi->rate_ || tsi->clr_id == -1)) {
183  // change CLR
184  tsi->clr_id = p->srcAddr_;
185  new_clr = 1;
186  }
187  if (tsi->clr_id == p->srcAddr_) {
188  if (adjusted_rate >= 0) {
189  tsi->expected_rate = adjusted_rate;
190  if (tsi->expected_rate > tsi->rate_)
191  TfmccSrcIncrRate (agent);
192  else
193  TfmccSrcDecrRate (agent);
194  }
195  }
196  }
197 
198  // determine which timestamp to include in data packet
199  if (tsi->clr_id == p->srcAddr_) {
200  // remember CLR timestamp to include in packets w/o any other RTT feedback
201  tsi->clr_timestamp = nck->timestamp;
202  tsi->clr_last_feedback = now;
203  if (! nck->have_rtt || new_clr)
204  cur_prio = TFMCC_SRC_PRIO_FORCE;
205  else
206  cur_prio = TFMCC_SRC_PRIO_CLR;
207  } else {
208  if (! nck->have_rtt)
209  cur_prio = TFMCC_SRC_PRIO_NO_RTT;
210  else
211  cur_prio = TFMCC_SRC_PRIO_RECV;
212  }
213  if (cur_prio > tsi->rtt_prio
214  || (cur_prio == tsi->rtt_prio && nck->rate < tsi->rtt_rate)) {
215  tsi->rtt_prio = cur_prio;
216  tsi->rtt_recv_id = p->srcAddr_;
217  tsi->rtt_recv_timestamp = nck->timestamp;
218  tsi->rtt_recv_last_feedback = now;
219  tsi->rtt_rate = nck->rate; // use reported rate for suppression
220  }
221 }

224 void TfmccSrcSlowStart (Agent * agent)
225 {
226  TfmccSrcInfo * tsi = (TfmccSrcInfo *) agent->info_;
227  tw_lp * lp = tw_getlp (agent->nodeId_);
228  double now = tw_now (lp);
229 
230  if (tsi->rate_ + TFMCC_SMALLFLOAT < tsi->pktSize_ / tsi->max_rtt_ ) {
231  /* if this is the first report, change rate to 1 per rtt */
232  /* compute delta so rate increases slowly to new value */
233  tsi->oldrate_ = tsi->rate_;
234  tsi->rate_ = tsi->fairsize_ / tsi->max_rtt_;
235  tsi->delta_ = (tsi->rate_ - tsi->oldrate_)
236  / (tsi->rate_ * tsi->max_rtt_ / tsi->pktSize_);
237  tsi->last_change_ = now;
238  } else {
239  /* else multiply the rate by ssmult_, and compute delta, */
240  /* so that the rate increases slowly to new value */
241  if (tsi->maxrate_ > 0) {
242  if (tsi->ssmult_ * tsi->rate_ < tsi->maxrate_
243  && now - tsi->last_change_ > tsi->max_rtt_) {
244  tsi->rate_ = tsi->ssmult_ * tsi->rate_;
245  tsi->delta_ = (tsi->rate_ - tsi->oldrate_)
246  / (tsi->rate_ * tsi->max_rtt_ / tsi->pktSize_);
247  tsi->last_change_ = now;
248  } else {
249  if (tsi->rate_ > tsi->maxrate_
250  || now - tsi->last_change_ > tsi->max_rtt_) {
251  tsi->rate_ = tsi->maxrate_;
252  tsi->delta_ = (tsi->maxrate_ - tsi->oldrate_)
253  / (tsi->rate_ * tsi->max_rtt_ / tsi->pktSize_);
254  if (tsi->delta_ < 0) tsi->delta_ = 0;
255  tsi->last_change_ = now;
256  }
257  }
258  } else {
259  if (now - tsi->last_change_ > tsi->max_rtt_) {
260  tsi->rate_ = tsi->ssmult_ * tsi->rate_;
261  tsi->delta_ = (tsi->rate_ - tsi->oldrate_)
262  / (tsi->rate_ * tsi->max_rtt_ / tsi->pktSize_);
263  tsi->last_change_ = now;
264  }
265  }
266  }

270 void TfmccSrcIncrRate (Agent * agent)
271 {
272  TfmccSrcInfo * tsi = (TfmccSrcInfo *) agent->info_;
273  tw_lp * lp = tw_getlp (agent->nodeId_);
274  double now = tw_now (lp), next;
275  double mult = (now - tsi->last_change_) / tsi->max_rtt_;
276  if (mult > 2) mult = 2;
277 
278  tsi->rate_ = tsi->rate_ + (tsi->fairsize_ / tsi->max_rtt_) * mult;
279  if (tsi->rate_ > tsi->expected_rate) tsi->rate_ = tsi->expected_rate;
280 
281  if (tsi->rate_ < tsi->fairsize_ / tsi->max_rtt_) {
282  tsi->rate_ = tsi->fairsize_ / tsi->max_rtt_;
283  }
284 
285  tsi->last_change_ = now;
286 
287  // check if rate increase should impact send timer
288  next = tsi->pktSize_ / tsi->rate_;
289  if (tsi->inter_packet >= 2 * next) {
290  assert(next > TFMCC_SMALLFLOAT);
291  ReschedTimer (agent, TFMCC_SRC_TIMER_SEND_PKT, next,
292  & (tsi->sendPktTimerEvnt_));
293  }

369 void TfmccSrcTimer (Agent * agent, int timerId, int serial)
370 {
371  TfmccSrcInfo * tsi = (TfmccSrcInfo *) agent->info_;
372  tw_lp * lp = tw_getlp (agent->nodeId_);
373  double now = tw_now (lp), xrate, max_variation, next = 0;
374 
375  if (! tsi->active_) return;
376 
377  switch (timerId) {
378  case TFMCC_SRC_TIMER_SEND_PKT:
379  xrate = -1;
380  tsi->sendPktTimerEvnt_ = NULL;
381 
382  // new feedback round every T = t_factor * max_RTT,
383  // if at least one non-CLR FBM was received (supp_rate_ < DBL_MAX)
384  // or after T = 2 * t_factor * max_RTT (only one receiver in group)
385  if ((now - tsi->round_begin_ >= tsi->t_factor_ * tsi->max_rtt_
386  && tsi->supp_rate_ < DBL_MAX)
387  || (now - tsi->round_begin_ >= 2 * tsi->t_factor_ * tsi->max_rtt_)) {
388  tsi->round_id = (tsi->round_id + 1) % TFMCC_MAX_ROUND_ID;
389  tsi->round_begin_ = now;
390  tsi->supp_rate_ = DBL_MAX;
391  // XXX draft says only inc when max_rtt unchanged from prev round
392  tsi->max_rtt_ *= TFMCC_SRC_MAX_RTT_DECAY;
393  if (tsi->max_rtt_ < tsi->pktSize_ / tsi->rate_) {
394  tsi->max_rtt_ = tsi->pktSize_ / tsi->rate_;
395  }
396  }
397 
398  // time out CLR
399  if (now - tsi->clr_last_feedback
400  > tsi->max_rtt_ * TFMCC_SRC_DEC_NO_REPORT) {
401  tsi->clr_id = -1;
402  }
403 
404  TfmccSrcSendPkt (agent);
405 
406  // during slow start, we increase rate
407  // slowly - by amount delta per packet
408  if ((tsi->rate_change_ == TFMCC_SRC_STATE_SLOW_START)
409  && (tsi->oldrate_ + TFMCC_SMALLFLOAT < tsi->rate_)) {
410  tsi->oldrate_ = tsi->oldrate_ + tsi->delta_;
411  xrate = tsi->oldrate_;
412  } else {
413  xrate = tsi->rate_;
414  }
415 
416  if (xrate > TFMCC_SMALLFLOAT) {
417  tsi->inter_packet = tsi->pktSize_ / xrate;
418 
419  // randomize between +-50% packet interval, but no more than max_rtt gap
420  max_variation = Max (0, Min (0.5 * tsi->inter_packet,
421  tsi->max_rtt_ - tsi->inter_packet));
422 
423  next = tsi->inter_packet - max_variation
424  + 2 * max_variation * (rand () / ((double) RAND_MAX));
425  assert (next > TFMCC_SMALLFLOAT);
426  tsi->sendPktTimerEvnt_ = ScheduleTimer
427  (agent, TFMCC_SRC_TIMER_SEND_PKT, next);
428  }
429  break;
430  case TFMCC_SRC_TIMER_NO_FDBK:
431  tsi->rate_ *= 0.5;
432  next = tsi->pktSize_ / tsi->rate_;
433  assert (next > TFMCC_SMALLFLOAT);
434  ReschedTimer
435  (agent, TFMCC_SRC_TIMER_SEND_PKT, next, & (tsi->sendPktTimerEvnt_));
436  tsi->noFdbkTimerEvnt_ = ScheduleTimer
437  (agent, TFMCC_SRC_TIMER_NO_FDBK, TFMCC_SRC_DEC_NO_REPORT * tsi->max_rtt_);
438  break;
439  }
